create or replace trigger restrictedCount after insert or update on payroll
for each row
declare

monthSal NUMBER;
countEmp NUMBER;

begin 
    Select count(esalary)
    into monthSal
    from payroll
    where esalary > 10000;
    
    if (monthSal > 3) then
        RAISE_APPLICATION_ERROR(-20001, 'cannot add more than 3 employees with more than 10000 monthly');
    end if;
end;
